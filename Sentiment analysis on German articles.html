def clean_words(data, stopwords):
    # categories = ["Etat", "Web", "International", "Wirtschaft", "Panorama",
    #             "Kultur", "Inland", "Sport", "Wissenschaft"]
    clean = dict()
    words = []
    # stemmer = GermanStemmer()
    i = 0
    # [A-Z]\w+;
    for article in data:
        a = re.split("[A-Z]\w+;", article)
        b = tokenize.word_tokenize(a[1])
        for word in b:
            # word = stemmer.stem(word)
            if word not in string.punctuation:
                if word.lower() not in stopwords:
                    if word not in words:
                        words.append(word)
        clean.__setitem__(str(i), words)
        i += 1
        words = []

    return clean

def calculate_sentiment(clean, positive, negative):
    pos = neg = neu = 0
    senti = []
    for words in clean.values():
        for word in words:
            for i in positive.keys():
                w = i.split(',')
                if w.__contains__(word):
                    # pos += positive.get(i)
                    pos += 1
                    break
            for i in negative.keys():
                w = i.split(",")
                if w.__contains__(word):
                    # neg += negative.get(i)
                    neg += 1
                    break
        senti.append(float((pos - neg) / len(words)))
    return senti

def get_neg_pos():
    negative = dict()
    positive = dict()

    with open("negative.txt", encoding="utf-8") as german_negative:
        for line in german_negative:
            x = line.split("|")
            y = x[1].split("\t")
            if y.__len__() == 3:
                y[2] = y[2].rstrip("\n")
                negative[x[0] + ',' + y[2]] = float(y[1])
            else:
                negative[x[0]] = float(y[1])

    with open("positive.txt", encoding="utf-8") as german_positive:
        for line in german_positive:
            x = line.split("|")
            y = x[1].split("\t")
            if y.__len__() == 3:
                y[2] = y[2].rstrip("\n")
                positive[x[0] + ',' + y[2]] = float(y[1])
            else:
                positive[x[0]] = float(y[1])

    return negative, positive
